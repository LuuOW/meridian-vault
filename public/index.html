Sign in from a terminal with an Ed25519 SSH key 
FP=$(ssh-keygen -lf ~/.ssh/id_ed25519.pub | awk '{print $2}')
CHAL=$(curl -sS -X POST -H 'Content-Type: application/json' \
  --data "{\"fingerprint\":\"$FP\"}" \
  https://vault.ask-meridian.uk/ssh/challenge)
NONCE=$(echo "$CHAL" | jq -r .nonce)
CID=$(echo "$CHAL" | jq -r .challenge_id)
SIG=$(printf '%s' "$NONCE" | ssh-keygen -Y sign -n vault.ask-meridian.uk -f ~/.ssh/id_ed25519)
curl -sS -X POST -H 'Content-Type: application/json' \
  --data "$(jq -n --arg cid "$CID" --arg sig "$SIG" '{challenge_id:$cid,signature:$sig}')" \
  https://vault.ask-meridian.uk/ssh/verify -c ~/.vault-cookies.txt

Your SSH key must be registered first (use the SSH keys panel after unlocking with the token).
